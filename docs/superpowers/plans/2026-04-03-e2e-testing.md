#!/usr/bin/env python3
"""Tauri 应用进程管理器：启动 debug build → 等待就绪 → 执行测试命令 → 关闭"""

import argparse
import os
import platform
import signal
import subprocess
import sys
import time
from pathlib import Path

PROJECT_ROOT = Path(__file__).resolve().parent.parent.parent
APP_NAME = "Dimkey"


def find_debug_binary() -> Path:
    """查找 Tauri debug build 产物"""
    system = platform.system()
    if system == "Darwin":
        binary = PROJECT_ROOT / "src-tauri" / "target" / "debug" / APP_NAME
        if not binary.exists():
            # macOS app bundle 路径
            bundle = PROJECT_ROOT / "src-tauri" / "target" / "debug" / "bundle" / "macos" / f"{APP_NAME}.app" / "Contents" / "MacOS" / APP_NAME
            if bundle.exists():
                return bundle
        return binary
    elif system == "Windows":
        return PROJECT_ROOT / "src-tauri" / "target" / "debug" / f"{APP_NAME}.exe"
    else:
        return PROJECT_ROOT / "src-tauri" / "target" / "debug" / APP_NAME.lower()


def build_debug():
    """构建 debug 版本"""
    print("[with_tauri] 构建 debug 版本...")
    # 先构建前端
    subprocess.run(["npm", "run", "build"], cwd=PROJECT_ROOT, check=True)
    # 再构建 Rust
    subprocess.run(
        ["cargo", "build"],
        cwd=PROJECT_ROOT / "src-tauri",
        check=True,
    )
    print("[with_tauri] 构建完成")


def wait_for_ready(proc: subprocess.Popen, timeout: int = 60):
    """等待应用启动就绪（轮询进程是否存活 + 等待窗口出现）"""
    start = time.time()
    while time.time() - start < timeout:
        if proc.poll() is not None:
            raise RuntimeError(f"应用启动失败，退出码: {proc.returncode}")
        # 简单等待 — Tauri 应用启动后 WebView 需要几秒加载
        time.sleep(2)
        print("[with_tauri] 应用已启动")
        return
    raise TimeoutError(f"应用在 {timeout} 秒内未启动")


def main():
    parser = argparse.ArgumentParser(description="Tauri E2E 测试运行器")
    parser.add_argument("--no-build", action="store_true", help="跳过构建步骤")
    parser.add_argument("--keep-alive", action="store_true", help="测试后不关闭应用")
    parser.add_argument("--timeout", type=int, default=60, help="启动超时秒数")
    parser.add_argument("command", nargs=argparse.REMAINDER, help="测试命令（-- 之后）")
    args = parser.parse_args()

    # 去掉 -- 分隔符
    cmd = args.command
    if cmd and cmd[0] == "--":
        cmd = cmd[1:]
    if not cmd:
        print("用法: python with_tauri.py [options] -- <test command>")
        print("示例: python with_tauri.py -- pytest e2e/tests/ -v")
        sys.exit(1)

    binary = find_debug_binary()

    # 构建
    if not args.no_build and not binary.exists():
        build_debug()

    if not binary.exists():
        print(f"[with_tauri] 未找到 debug binary: {binary}")
        print("[with_tauri] 请先运行: cargo build (在 src-tauri 目录)")
        sys.exit(1)

    # 设置测试环境变量
    env = os.environ.copy()
    env["DIMKEY_E2E"] = "1"
    # Tauri WebView 的 devUrl（debug build 使用本地前端资源，不需要 dev server）
    env["DIMKEY_TEST_BINARY"] = str(binary)

    # 启动应用
    print(f"[with_tauri] 启动应用: {binary}")
    proc = subprocess.Popen([str(binary)], env=env)

    try:
        wait_for_ready(proc, args.timeout)

        # 执行测试命令
        print(f"[with_tauri] 执行测试: {' '.join(cmd)}")
        result = subprocess.run(cmd, cwd=PROJECT_ROOT, env=env)

        if args.keep_alive:
            print("[with_tauri] --keep-alive: 应用保持运行，按 Ctrl+C 退出")
            proc.wait()

        sys.exit(result.returncode)

    except KeyboardInterrupt:
        print("\n[with_tauri] 收到中断信号")
    finally:
        if proc.poll() is None:
            print("[with_tauri] 关闭应用...")
            proc.send_signal(signal.SIGTERM)
            try:
                proc.wait(timeout=10)
            except subprocess.TimeoutExpired:
                proc.kill()
        print("[with_tauri] 完成")


if __name__ == "__main__":
    main()
```

"""Dimkey E2E 测试通用操作封装"""

from pathlib import Path
from playwright.sync_api import Page, expect

FIXTURES_DIR = Path(__file__).resolve().parent.parent / "fixtures"
OUTPUT_DIR = Path(__file__).resolve().parent.parent / "output"


def wait_for_view(page: Page, view_name: str, timeout: int = 30_000):
    """等待视图切换到指定状态
    
    view_name: empty | dropzone | processing | comparison | restore
    """
    page.wait_for_selector(f'[data-testid="view-{view_name}"]', timeout=timeout)


def import_file_via_picker(page: Page, file_path: str):
    """通过文件选择器导入文件
    
    使用 Playwright 的 filechooser 事件拦截 Tauri 的文件对话框
    注意：Tauri 使用原生文件对话框，Playwright 无法直接拦截。
    这里通过在 dropzone 上触发 click 并使用 set_input_files 来模拟。
    如果不可行，改用 Tauri IPC 直接调用 import_file 命令。
    """
    # Tauri 使用原生对话框，E2E 中通过 evaluate 调用 invoke 导入
    abs_path = str(Path(file_path).resolve())
    page.evaluate(f"""
        async () => {{
            const {{ invoke }} = window.__TAURI_INTERNALS__;
            const content = await invoke('import_file', {{ filePath: '{abs_path}' }});
            // 触发 store 更新 — 通过自定义事件通知前端
            window.dispatchEvent(new CustomEvent('e2e-file-imported', {{ detail: content }}));
        }}
    """)


def import_file_via_ipc(page: Page, file_path: str):
    """通过 Tauri IPC 直接导入文件（绕过原生文件对话框）
    
    这是 E2E 测试中导入文件的推荐方式。
    在 dropzone 视图中调用，会触发完整的检测流程。
    """
    abs_path = str(Path(file_path).resolve())
    # 使用 page.evaluate 调用前端的 store action
    page.evaluate(f"""
        async () => {{
            // 获取 zustand store — 通过 window 暴露（需要在测试模式下暴露）
            if (window.__DIMKEY_STORE__) {{
                await window.__DIMKEY_STORE__.getState().handleFileDrop(['{abs_path}']);
            }}
        }}
    """)


def wait_for_processing_done(page: Page, timeout: int = 60_000):

```

## Dimkey 视图状态机

```
empty ──(选择工作区)──→ dropzone
dropzone ──(导入文件)──→ processing
processing ──(完成)──→ comparison
comparison ──(返回)──→ dropzone
dropzone ──(AI还原/工作区还原)──→ restore
restore ──(返回)──→ dropzone
```

## 编写新测试用例

当用户描述测试场景时，按以下模式编写：

1. **确定起始视图** — 大多数测试从 `dropzone` 开始
2. **导入文件** — 通过 `page.evaluate` 调用 `window.__DIMKEY_STORE__`
3. **等待视图切换** — `wait_for_view` 或 `wait_for_processing_done`
4. **执行操作** — 使用 helpers 或 `page.click/fill/keyboard`
5. **断言结果** — 高亮数量、视图状态、元素可见性
6. **截图记录** — `take_diagnostic` 保存诊断信息

测试文件放在 `e2e/tests/` 目录，命名 `test_<场景>.py`，使用 `pytestmark` 标记优先级。

## 文件导入方式

Tauri 使用原生文件对话框，Playwright 无法拦截。测试中通过暴露的 store 直接触发：

```python
page.evaluate(f"""
    async () => {{
        const store = window.__DIMKEY_STORE__;
        if (store) await store.getState().handleFileDrop(['{file_path}']);
    }}
""")
```

前提：在 DEV 模式或 `DIMKEY_E2E=1` 时，store 已暴露到 `window.__DIMKEY_STORE__`。

## 常见问题排查

- **视图切换超时** — 检查应用是否启动、文件路径是否正确
- **高亮数为 0** — 检查 fixture 文件中是否包含敏感信息、检测引擎是否正常加载
- **store 未暴露** — 确认 DEV 模式或 DIMKEY_E2E 环境变量已设置
- **导出测试** — Tauri 原生保存对话框需要通过 IPC 直接调用 export_file
